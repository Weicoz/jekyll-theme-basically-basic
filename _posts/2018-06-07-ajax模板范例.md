### ajax模板范例


$.ajax({
   url: 'props_ajax.php?customer_id=<?php echo $customer_id_en;?>',
   data:{name:name,relation_type_id:relation_type_id,keyid:keyid},
   dataType: 'json',
   type: 'post',
   success:function(data){
       if(data.status == 1){
           $("#frm_pro").submit();
       }else{
           alert('提示信息：该分类下有相同的属性名！');
       }
   }
});
