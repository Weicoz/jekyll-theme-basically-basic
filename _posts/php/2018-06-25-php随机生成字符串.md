### php随机生成字符串

{% highlight php linenos %}
$chars = 'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
$password = "";
for ( $i = 0; $i < 8; $i++ ){
    $password .= $chars[ mt_rand(0, strlen($chars) - 1) ];
}
{% endhighlight %}
